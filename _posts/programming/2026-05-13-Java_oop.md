## <span style="color:#802548">_クラス分け：継承_</span>

- もし、switch文に沿って分岐するコードが多いと、あれはクラス分けがうまくいけなかったという信号だ

```java
class Figure {
    enum Shape {RECTANGLE, CIRCLE};

    final Shape shape;

    double length;
    double width;

    double radious;

    Figure(double radius) {
        shape = Shape.CIRCLE;
        this.radius = radius;
    }

    Figure(double length, double width) {
        shape = Shape.RECTANGLE;
        this.length = length;
        this.width = width;
    }

    double area() {
        switch(shape) {
            case RECTANGLE:
                return length * width;
            case CIRCLE:
                return Math.PI * (radius * radius);
            default :
                throw new AssertionError();
        }
    }
}
```

- 以下のようにクラス分けを行う
- 表現したいことが、ヒエラルキーに基づいたものであれば、それを継承を介して表す

```java
abstract class Figure {
    abstract double area();
}

class Circle extends Figure {
    final double radius;

    Circle(double radius) {
        this.radius = radius;
    }

    double area() {
        return Math.PIU * (radius * radius);
    }
}

class Rectangle extends Figure {
    final double length;
    final double width;

    Rectangle(double length, double width) {
        this.length = length;
        this.width = width;
    }

    double area() {
        return (length * width);
    }
}
```

```java
    int getMaxHP();
    void setHitPoint(int hp);

class GroundUnit extends Unit implements Repairable {

		Super(hp);

class AirUnit extends Unit implements Repairable {

    @Override public void setHitPoint(int hp) { this.hitPoint = hp; }

class Building implements Repariable {

		MAX_HP = hp;
	}

	@Override public int getHitPoint() { return this.hitPoint; }
    @Override public int getMaxHP() { return this.MAX_HP; }
    @Override public void setHitPoint(int hp) { this.hitPoint = hp; }
}

void repair(Repairable r) {
    while (r.getHitPoint() < r.getMaxHP()) {
        r.setHitPoint(r.getHitPoint() + 1);
```

```java
    private Waller wallet;

    public int getPayment(int payment) {
        if (wallet == null) {
            throw new NotEnoughMoneyException();
        }
        if (wallet.getTotalMoney() >= payment) {
            wallet.substractMoney(payment);
            
            return payment;
        }

        throw new NotEnoughMoneyException();
    }
}
```
